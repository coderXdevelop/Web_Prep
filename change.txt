    } else {
        section.style.display = "none";
    }
 }


6.Discount on Purchase >1000
script.js-->(add new function):
 function discountedTotal() {
    const total = cartTotal();

    if (total > 1000) {
        return total * 0.90; // 10% discount
    }
    return total;
 }
